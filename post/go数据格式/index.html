fmt.Println()

    //解析到interface
    tt := []byte(`{"name":"Sun","age":18,"Sex":"man"}`)
    var i 接口{}
    err = json.Unmarshal(tt,&i)
    if err != nil{
        fmt.Println(err)
    }
    //自动转到map
    m := i.(map[string]接口{})
    for k,v := range m{
        switch vv :=v.(type) {
        case int:
            fmt.Println(k,"是int类型",vv)
        case string:
            fmt.Println(k,"是string类型",vv)
        default:
            fmt.Println(k,"other")
        }
    }
}

func writejson(filename string)(err error)  {
    var person []*Person
    //假数据
    for i:=0;i<10;i++{
        p := &Person{
            Name:fmt.Sprintf("name%d",i),
            Age:rand.Intn(100),
            Sex:"male",
        }
        person = append(person,p)
    }
    //二进制json序列化
    data,err := msgpack.Marshal(person)
    if err != nil {
        fmt.Println(err)
        return
    }
    err = ioutil.WriteFile(filename,data,0666)
    if err != nil {
        fmt.Println(err)
        return
    }
    return
}
